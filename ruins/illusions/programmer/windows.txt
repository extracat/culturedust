var timeout;
var pos=25;
function moveup()
{          
      if (pos < 25) pos+=2;
      document.all.t.style.top=pos;   
      clearTimeout(timeout);
      timeout=setTimeout("moveup()",5);
}

function movedown()
{           
      if (pos > -300) pos-=2;
      document.all.t.style.top=pos;       
      clearTimeout(timeout);
      timeout=setTimeout("movedown()",5);
}

function stop()
{           
      clearTimeout(timeout);
}



</SCRIPT>
